    %% Domain Layer : Core business
    namespace Domain {
        class Order {
            -Long id
            -Long subscriberId
            -LocalDateTime orderDate
            -String deliveryAddress
            -LocalDate deliveryDate
            -Double totalPrice
            -List~OrderLine~ lines
            +calculateTotalPrice() void
        }
        class OrderLine {
            -Long menuId
            -String menuName
            -Integer quantity
            -Double unitPrice
            -Double linePrice
        }
    }

    %% Application Layer : Business logic
    namespace Application {
        class OrderRepository {
            <<interface>>
            +save(Order order) Order
            +findAll() List~Order~
            +findById(Long id) Optional~Order~
            +deleteById(Long id) boolean
        }
        class OrderService {
            -OrderRepository repository
            +createOrder(OrderInput input) Order
            +listOrders(Long subscriberId) List~Order~
            +findOrder(Long id) Optional~Order~
            +deleteOrder(Long id) boolean
            +updateOrder(Long id, OrderUpdateInput input) Optional~Order~
        }
        class OrderInput {
            +Long subscriberId
            +String deliveryAddress
            +String deliveryDate
            +List~OrderLineInput~ lines
        }
        class OrderUpdateInput {
            +String deliveryAddress
            +String deliveryDate
        }
        class OrderLineInput {
            +Long menuId
            +Integer quantity
        }
    }

    %% Infrastructure Layer : Adapters (API and DB)
    namespace Infrastructure {
        class OrderResource {
            -OrderService service
            +getAll(Long subscriberId) List~Order~
            +create(OrderInput input) Response
            +getById(Long id) Response
            +update(Long id, OrderUpdateInput input) Response
            +delete(Long id) Response
        }
        class JdbcOrderRepository {
            -String url
            -String user
            -String password
            -loadConfig() Properties
            -getConnection() Connection
            +save(Order order) Order
            +findAll() List~Order~
            +findById(Long id) Optional~Order~
            +deleteById(Long id) boolean
        }
    }

    %% Relations
    Order "1" *-- "*" OrderLine : contains
    OrderInput "1" *-- "*" OrderLineInput : contains
    
    %% Injection et implémentation
    OrderService --> OrderRepository : uses
    JdbcOrderRepository ..|> OrderRepository : implements
    OrderResource --> OrderService : delegates
    
    %% Dépendances aux DTOs
    OrderService ..> OrderInput : reads
    OrderService ..> OrderUpdateInput : reads

    %% Dépendances au Domaine (Nouveaux liens)
    OrderService ..> Order : manipulates
    OrderRepository ..> Order : handles
    OrderResource ..> Order : returns
